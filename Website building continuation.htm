// src/pages/MediaPage.jsx
import { useState } from "react";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";

export default function MediaPage() {
  const [supporters, setSupporters] = useState([]);
  const [message, setMessage] = useState("");
  const [file, setFile] = useState(null);

  const handleSupport = (e) => {
    e.preventDefault();

    if (!message && !file) return;

    const newSupport = {
      text: message || "",
      file: file ? URL.createObjectURL(file) : null,
      type: file ? file.type : "text",
    };

    setSupporters([newSupport, ...supporters]);
    setMessage("");
    setFile(null);
  };

  return (
    <div className="p-6 max-w-6xl mx-auto space-y-10">
      {/* Featured Video */}
      <section>
        <h1 className="text-3xl font-bold mb-4">Campaign Media</h1>
        <div className="rounded-2xl overflow-hidden shadow-lg">
          <iframe
            className="w-full aspect-video"
            src="https://www.youtube.com/embed/dQw4w9WgXcQ"
            title="Featured Campaign Video"
            allowFullScreen
          ></iframe>
        </div>
      </section>

      {/* Video Grid */}
      <section>
        <h2 className="text-2xl font-semibold mb-4">Past Events & Highlights</h2>
        <div className="grid sm:grid-cols-2 md:grid-cols-3 gap-6">
          {["abc123", "def456", "ghi789"].map((id, idx) => (
            <Card key={idx} className="rounded-2xl shadow-md">
              <CardContent className="p-2">
                <iframe
                  className="w-full aspect-video rounded-xl"
                  src={`https://www.youtube.com/embed/${id}`}
                  title={`Event ${idx + 1}`}
                  allowFullScreen
                ></iframe>
              </CardContent>
            </Card>
          ))}
        </div>
      </section>

      {/* Voice Your Support */}
      <section>
        <h2 className="text-2xl font-semibold mb-4">Voice Your Support</h2>
        <form
          onSubmit={handleSupport}
          className="flex flex-col gap-3 mb-6 border p-4 rounded-xl shadow-sm"
        >
          <textarea
            className="w-full border rounded-xl px-4 py-2"
            placeholder="Write your message..."
            rows={3}
            value={message}
            onChange={(e) => setMessage(e.target.value)}
          ></textarea>

          <input
            type="file"
            accept="audio/*,video/*"
            onChange={(e) => setFile(e.target.files[0])}
          />

          <Button type="submit" className="self-start">
            Submit Support
          </Button>
        </form>

        {/* Supporter Wall */}
        <div className="space-y-4">
          {supporters.length === 0 ? (
            <p className="text-gray-500">No supporter messages yet. Be the first!</p>
          ) : (
            supporters.map((s, idx) => (
              <Card key={idx} className="rounded-xl shadow-sm">
                <CardContent className="p-3 space-y-3">
                  {s.text && <p>{s.text}</p>}
                  {s.file && s.type.startsWith("audio") && (
                    <audio controls src={s.file} className="w-full" />
                  )}
                  {s.file && s.type.startsWith("video") && (
                    <video controls src={s.file} className="w-full rounded-xl" />
                  )}
                </CardContent>
              </Card>
            ))
          )}
        </div>
      </section>
    </div>
  );
}
